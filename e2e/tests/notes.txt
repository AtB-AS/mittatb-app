List all text attributes of a type
*********************************
(Matching does not support RegExp)

    await element(by.type('RCTTextView')).getAttributes()
        .then(e => {
          if ('elements' in e) {
            for (let elem of e.elements) {
              console.log('RCTTextView: ', elem.text)
            }
          }
          if (!('elements' in e)) {
              console.log('Elem.text: ', e.text)
              console.log('Elem.value: ', e.value)
              console.log('Elem.enabled: ', e.enabled)
              console.log('Elem.label: ', e.label)
              console.log('Elem.placeholder: ', e.placeholder)
          }
        })


Element matcher examples
************************

//Multiple elements found for “MATCHER((class ⊇ “RCTTextView” && (text == “Trondheim S”)) && ANCESTOR(class ⊇ “RCTView” && ANCESTOR(class ⊇ “RCTView” && DESCENDANT(label == “Bussholdeplass”))))”
    await element(
        by.type('RCTTextView').and(by.text('Trondheim S'))
            .withAncestor(by.type('RCTView')
                .withAncestor(by.type('RCTView')
                    .withDescendant(by.label('Bussholdeplass')
                    )
                )
            )
    )
        .tap()

    //Multiple elements found for “MATCHER((class ⊇ “RCTView” && DESCENDANT(label == “Trondheim S Trondheim”)) && class ⊇ “RCTView” && DESCENDANT(label == “Bussholdeplass”))”
    await element(
        by.type('RCTView')
            .withDescendant(
                by.label('Trondheim S Trondheim')
            )
            .and(by.type('RCTView')
                .withDescendant(
                    by.label('Bussholdeplass')
                )
            )
    )
        .tap()

    await element(
        by.text('Trondheim S'))
        .atIndex(0)
        .tap()


    await element(
        by.type('RCTView').and(by.label('Bussholdeplass'))
            .withAncestor(by.label('Trondheim S, Trondheim ')
                .withDescendant(
                    by.label('Trondheim S Trondheim')
                )
            )
    )
        .tap()

   //Multiple elements found for “MATCHER((class ⊇ “RCTView” && DESCENDANT(label == “Trondheim S Trondheim”)) && DESCENDANT(label == “Bussholdeplass”))”
    await element(
        by.type('RCTView')
            .withDescendant(
                by.label('Trondheim S Trondheim')
            )
            .withDescendant(
                by.label('Bussholdeplass')
            )
    )
        .tap()

    //No elements found for “MATCHER(class ⊇ “RCTView” && DESCENDANT((label == “Trondheim S Trondheim”) && DESCENDANT(label == “Bussholdeplass”)))”
    await element(
        by.type('RCTView')
            .withDescendant(
                by.label('Trondheim S Trondheim')
                    .withDescendant(
                        by.label('Bussholdeplass')
                    )
            )
    )
        .tap()
